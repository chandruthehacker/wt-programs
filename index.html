2. Login Validation using Javascript

▶ Run Program 
<html>
  <body>
    <form>
      Username: <input type="text" id="u" /><br /><br />
      Password: <input type="password" id="p" /><br /><br />
      <input type="submit" onclick="return check();" />
    </form>

    <script>
      function check() {
        var u = document.getElementById("u").value;
        var p = document.getElementById("p").value;

        if (u == "") {
          alert("Enter Username");
          return false;
        }
        if (p.length < 6) {
          alert("Password must be 6 characters");
          return false;
        }
        alert("Login Success");
        return true;
      }
    </script>
  </body>
</html>

6. Profile Page

▶ Run Program 
<html>
  <head>
    <title>Profile Page</title>
  </head>
  <body>
    <h2>Profile Page</h2>

    <form>
      Name: <input type="text" /><br /><br />
      Email: <input type="email" /><br /><br />

      Gender:
      <input type="radio" name="g" /> Male
      <input type="radio" name="g" /> Female<br /><br />

      Date of Birth: <input type="date" /><br /><br />

      Upload Photo:
      <input type="file" onchange="showImage(event)" /><br /><br />

      <img id="img" width="120" /><br /><br />

      Address:<br />
      <textarea></textarea><br /><br />

      <input type="submit" value="Submit" />
      <input type="reset" value="Clear" />
    </form>

    <script>
      function showImage(event) {
        var img = document.getElementById("img");
        img.src = URL.createObjectURL(event.target.files[0]);
      }
    </script>
  </body>
</html>

9. Leap Year Checker

▶ Run Program 
<html>
  <body>
    <script>
      var year = prompt("Enter year");

      if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
        document.write(year + " is a Leap Year");
      } else {
        document.write(year + " is not a Leap Year");
      }
    </script>
  </body>
</html>
